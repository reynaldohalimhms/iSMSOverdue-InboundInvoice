select distinct top 20 
inv.invoicelegalnumber, inv.invoiceissuedate,
so.SalesOrderConfirmationNumber, so.SalesOrderCode
from SMSIndonesia_SUB_QAS.dbo.Invoice inv
join SMSIndonesia_SUB_QAS.dbo.SalesOrder so on so.salesorderid = inv.invoice_salesorderid
where inv.InvoiceLegalNumber like '%JOW%'
order by inv.invoiceissuedate desc;

SELECT top 100
        inv.InvoiceLegalNumber,
        so.SalesOrderCode,
        so.SalesOrderConfirmationNumber,
        so.SalesOrderStatus,
        so.SalesOrderIsRequestForCancellation,
        so.SalesOrderModDate
    FROM SMSIndonesia_SUB_QAS.dbo.SalesOrder so
    LEFT JOIN SMSIndonesia_SUB_QAS.dbo.Invoice inv 
        ON so.SalesOrderId = inv.Invoice_SalesOrderId
    WHERE so.salesorderconfirmationnumber like '%212%' order by so.SalesOrderConfirmationNumber asc;
